enum State { IDLE, PATROL, CHASE }

func _physics_process(delta: float) -> void:
    if not is_on_floor():
        velocity.y += gravity * delta

    match current_state:
        State.IDLE:
            process_idle(delta)
        State.PATROL:
            process_patrol(delta)
        State.CHASE:
            process_chase(delta)

    move_and_slide()
